#include "tp0.h"

void swap(int *x, int *y){
        int aux;
        aux = *x;
        *x = *y;
        *y = aux;
}

int maximo(int vector[],int n){
	if(n==0){
		return -1;
	}
	else if(n==1){
		return 0;
	}
	else{
		int max = 0;
		int i;
		for(i=0;i<n;i++){
			if(vector[i]>vector[max]){
				max = i;
			}
		}
		return max;
	}
}

int comparar(int vector1[], int n1, int vector2[], int n2) {
    if(n1 == 0 && n2 == 0){
        return 0;
    }
    else if(n1 == 0 && n2 > 0){
        return -1;
    }
    else if(n1 > 0 && n2 == 0){
        return 1;
    }
    else{
        int i = 0;
        while(i<n1 || i<n2){
            if (vector1[i]!=vector2[i]){
                if(vector1[i]<vector2[i]){
                    return -1;
                }
                else{
                    return 1;
                }
            }
            i++;
        }
        if(n1<n2){
            return -1;
        }
        else if (n1>n2){
            return 1;
        }
        else{
            return 0;
        }
    }
}

void seleccion(int vector[], int n){
        int fin = n;
        while(fin > 0){
            int pos_max = maximo(vector, fin);
            swap(&vector[pos_max],&vector[fin-1]);
            fin--;
        }
}
