#include <stdio.h>
#include <string.h>

int maximo(int vector[],int n){
	if(n==0){
		return -1;
	}
	else if(n==1){
		return 0;
	}
	else{
		int max = 0;
		int i;
		for(i=0;i<n;i++){
			if(vector[i]>vector[max]){
				max = i;
			}
		}
		return max;
	}
}

int main(){
	int vector[] = { -2000, -1500, -1000, -3000 };
	int largo = 4;
	int resultado = maximo(vector,largo);
	printf("El maximo se encuentra en la posicion %d",resultado);
	return 0;
}
